//%attributes = {"lang":"en","invisible":true} comment added and reserved by 4D.// --[HEADERSTART]-------------------------------------// User name (OS): jfletcher// Date and time: 04/05/12, 16:14:59// ----------------------------------------------------// Description://   Wrapper method called by CE_MON_Process.////   I moved this to a separate method (from CE_MON_Process)//   so that changes can be made at run time without//   restarting the monitor.// --[HEADEREND]---------------------------------------C_LONGINT($0;$error_l)C_LONGINT($currentMethodStamp_l)ARRAY TEXT($methodsToSave_at;0)// Get the last method stamp.$currentMethodStamp_l:=<>CE_LastMethodStamp_l// Get changed methods since last stamp.METHOD GET PATHS(Path All objects ;$methodsToSave_at;$currentMethodStamp_l;*)// Validate method paths (invalid paths will be removed).$error_l:=CE_METHOD_ValidateAll (->$methodsToSave_at)If ($error_l=<>ERR_METH_BadPathsFound_l)// Export the list.// Warn user, but this error is ignorable.// Launch this in a separate process, it shouldn't stop the method export.$error_l:=0End if // Save valid methods.If ($error_l=0)$error_l:=CE_METHOD_SaveAll (->$methodsToSave_at)End if // Check to see if the stamp changed and save it if so.If ($error_l=0)If (<>CE_LastMethodStamp_l#$currentMethodStamp_l)$error_l:=CE_EXT_PrefSet (<>CE_StampLabel_t;String($currentMethodStamp_l))If ($error_l=0)// Only update the stamp if the save was successful.<>CE_LastMethodStamp_l:=$currentMethodStamp_lEnd if End if End if $0:=$error_l
