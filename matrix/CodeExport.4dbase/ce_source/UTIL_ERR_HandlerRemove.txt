//%attributes = {"lang":"en","invisible":true} comment added and reserved by 4D.// --[HEADERSTART]-------------------------------------// User name (OS): jfletcher// Date and time: 04/05/12, 16:25:45// ----------------------------------------------------// Description://   Uninstall generic error handler, restore previous//   error handler.//// --[HEADEREND]---------------------------------------C_LONGINT($0;$error_l)// Note: these proc vars are set in UTIL_ERR_HandlerInstallC_TEXT(currentErrorHandler_t;genericErrorHandler_t)C_TEXT($installedErrorHandler_t)$installedErrorHandler_t:=Method called on error// Don't mess with it unless our error handler is installed.If ($installedErrorHandler_t=genericErrorHandler_t)ON ERR CALL(currentErrorHandler_t)// If our error executed, UTIL_ERR_4DError_l will have the error code.If (UTIL_ERR_4DError_l#0)$error_l:=<>ERR_4D_lEnd if End if $0:=$error_l
